CREATE DATABASE tvrec;

CREATE TABLE USUARIO(
   CORREO 			TEXT PRIMARY KEY,
   NOMBRE       	TEXT 					 NOT NULL,
   CONTRASENA  		TEXT 					 NOT NULL
);

CREATE TABLE TELEVISOR(
   MODELO 				TEXT PRIMARY KEY,   
   OBSERVACIONES		TEXT,
   MARCA           		TEXT,
   PRECIO          		REAL,
   TAMANOPANTALLA       TEXT,
   RESOLUCION         	TEXT,
   TIPODISPLAY			TEXT,
   urlwalmart				TEXT,
   URLBB				TEXT,
   calificaciOnwalmart		REAL,
   ACTIVO				BOOLEAN				 NOT NULL
   Datos_otra_tabla	TEXT
   
);

CREATE TABLE TELEVISORWALMART(
   MODELO 				TEXT PRIMARY KEY,
   MARCA           		TEXT,
   PRECIO          		REAL,
   TAMANOPANTALLA       TEXT,
   RESOLUCION         	TEXT,
   TIPODISPLAY			TEXT,
   urlwalmart				TEXT,
   calificaciOnwalmart		REAL,
   ACTIVO				BOOLEAN				 NOT NULL
);

CREATE TABLE TELEVISORBB(
   MODELO 				TEXT PRIMARY KEY,
   MARCA           		TEXT,
   PRECIO          		REAL,
   TAMANOPANTALLA       TEXT,
   RESOLUCION         	TEXT,
   TIPODISPLAY			TEXT,
   URLBB				TEXT,
   CALIFICACIONBB		REAL,
   ACTIVO				BOOLEAN				 NOT NULL
);

CREATE TABLE Calificacion(
   CORREO 				TEXT REFERENCES USUARIO(CORREO),
   MODELO           	TEXT REFERENCES TELEVISOR(MODELO),
   CALIFICACIONUSUARIO  REAL,
   PRIMARY KEY (CORREO,MODELO)
);

CREATE TABLE Precision(
   ID     SERIAL    PRIMARY KEY,
   MODELO TEXT,
   CALIFICACION REAL,
   SISTEMA_RECOMENDACION TEXT
);
